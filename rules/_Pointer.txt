EQUALS(u,w) :- PTR(u,v,0), DEREFFROM(w,v)
EQUALS(u,v) :- PTR(x,v,s), DEREFFROM(w,v), ARRAYSET(x,s,u)
CORRECT_PTR(v) :- PTR(u,v,w), GREATER_EQUALS(w,0),  SIZEOF(u,s), LESS(w,s)
PTR (u, v, m) :- SUM (v, w, m), PTR(u, w, 0)
INCORRECT_PTR(v) :- ZERO(v)
NONZERO(v) :- PTR(u, v, w)
SIZEOF_LESS(u, w) :- SIZEOF(u, v), LESS(v, w)
SIZEOF_GREATER_EQUALS(u, w) :- SIZEOF(u, v), GREATER_EQUALS(v, w)
