EQUALS(a, 0) :- ZERO(a)
ZERO(a) :- EQUALS(a, 0)
NOT_EQUALS(a,0) :- NONZERO(a)
NONZERO(a) :- NOT_EQUALS(a,0)
NOT(a,b) :- NOT(b,a)
ZERO(b) :- NOT(a,b), NONZERO(a)
NONZERO(b) :- NOT(a, b), ZERO(b)
OR(a,b,c) :- OR(a,c,b)
ZERO(b) :- OR(a,b,c), ZERO(a)
NONZERO(b) :- OR(a, b, c), NONZERO(a), ZERO(c) 
AND(a, b, c) :- AND(a, c, b)
ZERO(b) :- AND(a, b, c), ZERO(a), NONZERO(c)
NONZERO(b) :- AND(a, b, c), NONZERO(a)
